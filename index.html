Alpha-Beta pruning using JavaScript

Code:

let depth, nodeIndex, maximizingPlayer, alpha, beta;
const min = -1000, max = 1000;
function minimax(depth, nodeIndex, maximizingPlayer, values, alpha, beta){
if (depth == 3) {
return values[nodeIndex];
}
if(maximizingPlayer){
let best = min;
for (let i = 0; i < 2; i++) {
let val = minimax(depth + 1, nodeIndex * 2 + i, false, values, alpha, beta);
best = Math.max(best, val);
alpha = Math.max(alpha, best);
if(beta <= alpha){
break
}
}
return best;
}
else{
let best = max;
for (let i = 0; i < 2; i++) {
val = minimax(depth + 1, nodeIndex * 2 + i, true, values, alpha, beta);
best = Math.min(best, val);
beta = Math.min(beta, best);
if(beta <= alpha){
break
}
}
return best;
}
}
var values = [3, 5, 6, 9, 1, 2, 0, -1]
console.log("The optimal value is: " + minimax(0, 0, true, values, min, max))
